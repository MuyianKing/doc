~~~
uni-app
|── public //公共资源目录
|── src
|   |── api 
|   |   |── index.js 统一管理接口
|   |── components 公共组件
|   |── default 公共待选项
|   |── model 实体类
|   |── pages 页面
|   |── server
|   |── static 资源目录
|   |── store Vuex
|   |── uni_modules 第三方组件
|   |── util 公共js代码
|── App.vue
|── main.js
|── env.js 环境变量
|── uni.scss css变量
|── alias.config.js 别名配置
